        function mudaCorTitulo(novaCor) {
            // var serve para criar uma variável
            var element = document.getElementById("texto");
            // elemento = frase (onde está o id=texto)
            element.style.color = novaCor;
        }

        function formatacaoTitulo() {
            titulo = document.getElementById("tituloPrincipal");
            titulo.style.backgroundColor = "#ff0000";
            titulo.style.border = "5px solid #bb00ff";
            titulo.style.fontFamily = "Tahoma";
            titulo.style.fontSize = "14px";

            camposDeEntrada = document.getElementsByName("titulo1");
            camposDeEntrada[0].style.backgroundColor = "#FF0000";
            camposDeEntrada[1].style.border = "4px solid #FF000";
            camposDeEntrada[2].style.fontSize = "30px";

            frases = document.getElementsByClassName("subTitulo");
            frases[0].style.fontWeight = "bold";
            frases[1].style.fontStyle = "italic";
        }
        
        // uma variável pode ser global caso seja criada fora das functions

        function init() {
            formatacaoTitulo();
        }
        window.onload = init;
    </script>
